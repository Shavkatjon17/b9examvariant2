1-Method darajasida: 
@PostMapping("save")
@ModelAttribute("user")

return "response";
}
2-Method parametr darajasida
PostMapping("save")
Public String save (@ModelAttribute("user") User user, Model model) {
model.addAtribute("user", user);
return "response";
